Detailed Description

This interface holds the execution results of an IntelliBlink algorithm tool.

Usually you get the LIBAlgoResult object from ILIBTaskResult. You can then use this interface to access individual result.

Example Code

C++
ILIBTaskResultPtr taskResult;
long err = m_ib_service->ExecuteTaskSync(m_ib_task_id, image, &taskResult);
if (err != LPVErrorCode::LPVNoError) return; // execution failed
ILIBAlgoResultPtr outputResult = taskResult->AlgoResult[L"algorithm name"];
int dataCount = outputResult->Count();
for (int i = 0; i < dataCount; ++i) {
ILIBDataPtr ibData = outputResult->Item(i);
}
C#
LIBTaskResult taskResult;
int err = m_ib_service.ExecuteTaskSync(m_ib_task_id, image, out taskResult);
if (err != (int)LPVErrorCode.LPVNoError) return; // execution failed
LIBAlgoResult outputResult = taskResult.AlgoResult["algorithm name"];
int dataCount = outputResult.Count();
for (int i = 0; i < dataCount; ++i) {
LIBData ibData = outputResult.Item(i);
}

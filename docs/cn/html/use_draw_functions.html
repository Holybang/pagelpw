绘制函数

对于 LPW 图像类以及其他实现了 ILDrawable 接口的类型，我们通常都会提供绘制函数，用于绘制对象的内容到指定的设备，也就是输出的 HDC 句柄。

绘制函数相比于 LDisplay Control，是一种更为灵活和轻量的 API，帮助用户完全控制应用程序的绘制行为和流程。

以下为一些绘制函数的接口例子，包括图像、区域和检测结果等：

void ILImage::Draw (HDC hdc, double zoomX, double zoomY, double panX, double panY);
void ILRegion::Draw (HDC hdc, BOOL bHandles, double zoomX, double zoomY, double panX, double panY);
void ILMatchResults::Draw (HDC hdc, int index, LPVPatDrawFlags drawFlags, double zoomX, double zoomY, double panX, double panY)
void ILMatchResults::DrawAll (HDC hdc, LPVPatDrawFlags drawFlags, double zoomX, double zoomY, double panX, double panY)

每个绘制函数通常有以下通用参数，用于指定绘制的 HDC 句柄以及视窗。

参数
[in]hdc绘制设备，可为窗口、位图等的上下文句柄。
[in]zoomXx 方向的缩放系数，默认为真实尺寸 100。
[in]zoomYy 方向的缩放系数，默认设置为 0，即与 x 方向的缩放系数相同，也就是等比缩放。
[in]panXx 方向的平移。最终的 x 坐标按照该公式计算: \( (x + panX) \times zoomX \)
[in]panYy 方向的平移。最终的 y 坐标按照该公式计算: \( (y + panY) \times zoomY \)

每个绘制函数可能有其他额外的参数，用于控制绘制内容和行为。

示例代码

在 Bitmap 上绘制

C++
void DrawImageAndResultsOnStatic(CStatic* pic_item, ILImagePtr img, ILRectRegionPtr roi, ILMatchResultsPtr matchResults)
{
// 在绘制之前，需要先修改 Static 控件的风格以接受 Bitmap 内容
// 推荐在窗口的初始化方法 OnInitDialog() 中进行，如：
// ((CStatic*)GetDlgItem(IDC_IMG_PIC))->ModifyStyle(0xF, SS_BITMAP, SWP_NOSIZE);
CRect picRect;
pic_item->GetClientRect(picRect);
// 适配窗口尺寸缩放
double zoomx = (double)picRect.Width() / img->Width;
double zoomy = (double)picRect.Height() / img->Height;
CDC* picdc = pic_item->GetDC();
if (!picdc) return;
// 创建用于绘制的内存 DC 和 Bitmap
HDC pichdc = picdc->GetSafeHdc();
HDC hdc = CreateCompatibleDC(pichdc);
HBITMAP bitmap = CreateCompatibleBitmap(pichdc, picRect.Width(),
picRect.Height());
// 替换 DC 中的 Bitmap 为新创建的 Bitmap
HBITMAP prevBitmap = (HBITMAP)SelectObject(hdc, bitmap);
if (prevBitmap)
DeleteObject(prevBitmap);
// 刷新白色背景
FillRect(hdc, &picRect, (HBRUSH)GetStockObject(WHITE_BRUSH));
// 调用绘制函数，在 Bitmap 上绘制图像、区域和匹配结果
// 使用 0, 0 位移
img->Draw(hdc, zoomx, zoomy, 0, 0);
roi->Draw(hdc, VARIANT_TRUE, zoomx, zoomy, 0, 0);
matchResults->DrawAll(hdc, LPVPatDrawDefault, zoomx, zoomy, 0, 0);
// 将 Bitmap 刷新到窗口
BitBlt(pichdc, picRect.left, picRect.top, picRect.Width(), picRect.Height(),
hdc, 0, 0, SRCCOPY);
pic_item->ReleaseDC(picdc);
// 释放内存 DC 和 Bitmap
DeleteObject(hdc);
DeleteObject(bitmap);
}
C#
void drawImageAndResultsOnStatic(PictureBox pic_item, LImage img, LRectRegion roi, LMatchResults match_results)
{
// 适配窗口尺寸缩放
double zoomX = (double)pic_item.Width / img.Width;
double zoomY = (double)pic_item.Height / img.Height;
// 创建用于绘制的内存 DC 和 Bitmap
Bitmap bmp = new Bitmap(pic_item.Width, pic_item.Height);
Graphics gfx = Graphics.FromImage(bmp);
gfx.Clear(Color.White); // 刷新白色背景
IntPtr hdc = gfx.GetHdc(); // 获取 HDC 句柄
// 调用绘制函数，在 Bitmap 上绘制图像、区域和匹配结果
// 使用 0, 0 位移
img.Draw(hdc, zoomX, zoomY);
roi.Draw(hdc, true, zoomX, zoomY);
match_results.DrawAll(hdc,
LPVPatDrawFlags.LPVPatDrawDefault,
zoomX, zoomY);
gfx.ReleaseHdc(hdc);
gfx.Dispose();
// 将 Bitmap 设置到显示控件
if (pic_item.Image != null)
pic_item.Image.Dispose();
pic_item.Image = bmp;
}

在 OnPaint 方法中绘制

C++
void CMFCDlg::OnPaint()
{
// 其他的一些绘制逻辑
CPaintDC dc(this); // 获取绘制 DC
HDC hdc = dc.GetSafeHdc();
CRect picRect;
GetClientRect(picRect);
// 适配窗口尺寸缩放
double zoomx = (double)picRect.Width() / m_img->Width;
double zoomy = (double)picRect.Height() / m_img->Height;
// 刷新白色背景
FillRect(hdc, &picRect, (HBRUSH)GetStockObject(WHITE_BRUSH));
// 调用绘制函数，在 Bitmap 上绘制图像、区域和匹配结果
// 使用 0, 0 位移
m_img->Draw(hdc, zoomx, zoomy, 0, 0);
m_roi->Draw(hdc, VARIANT_TRUE, zoomx, zoomy, 0, 0);
m_matchResults->DrawAll(hdc, LPVPatDrawDefault, zoomx, zoomy, 0, 0);
// 其他的一些绘制逻辑
CDialogEx::OnPaint();
// 提示，当需要绘制的内容改变时：
// 调用 Invalidate() 请求一次重绘，这将有系统的消息循环处理，可能发生延时。单次循环内的多次请求也将被合并。
// 调用 UpdateWindow() 强制马上进行一次重绘。当你需要马上看到更新结果时使用。
// 你可以通过一些标志位来控制某些感兴趣的 @projectname 对象进行绘制，同时跳过一些。
// 你可能需要用到双缓存技术，以避免窗口刷新时发生的闪烁现象，方法是先建立一个内存中的 DC 并将结果绘制上去，而后将整个内存 DC 复制到目标 DC
}

// 获取绘制 DC
IntPtr hdc = e.Graphics.GetHdc();

// 调用绘制函数，在 Bitmap 上绘制图像、区域和匹配结果
// 使用 0, 0 位移

m_roi.Draw(hdc, true, zoomX, zoomY);
m_match_results.DrawAll(hdc,
LPVPatDrawFlags.LPVPatDrawDefault,
zoomX, zoomY);
// 最后，释放 DC
e.Graphics.ReleaseHdc(hdc);
// 提示，当需要绘制的内容改变时：
// 调用 Invalidate() 请求一次重绘，这将有系统的消息循环处理，可能发生延时。单次循环内的多次请求也将被合并。

// 你可能需要开启窗口的双缓存功能，以避免窗口刷新时发生的闪烁现象
}
